import json
import re

def extract_entities(text):
    entities = {
        "Crops": [],
        "Soil": [],
        "Location": [],
        "Time Statement": []
    }

    # Crops
    crops = ["Biomasse", "Holz", "holzbasierte Produkte", "Schnittholz", "Textilfaser"]
    for crop in crops:
        matches = re.finditer(r'\b' + re.escape(crop) + r'\b', text)
        for match in matches:
            entity = {
                "cropSpecies": {
                    "value": match.group(),
                    "span": [match.start(), match.end()]
                }
            }
            entities["Crops"].append(entity)

    # Soil
    soil = []
    matches = re.finditer(r'\b(soil|Boden)\b', text)
    for match in matches:
        entity = {
            "soilTexture": {
                "value": match.group(),
                "span": [match.start(), match.end()]
            }
        }
        soil.append(entity)

    # Location
    locations = ["nova-Institut", "Thünen-Institut für Waldwirtschaft", "FAO", "World Bank", "FRA", "IIASA-SSP", "WTO", "Germany"]
    for location in locations:
        matches = re.finditer(r'\b' + re.escape(location) + r'\b', text)
        for match in matches:
            if location == "Germany":
                entity = {
                    "country": {
                        "value": match.group(),
                        "span": [match.start(), match.end()]
                    }
                }
            else:
                entity = {
                    "region": {
                        "value": match.group(),
                        "span": [match.start(), match.end()]
                    }
                }
            entities["Location"].append(entity)

    # Time Statement
    time_statements = ["2020", "2050", "2023", "2024"]
    for time_statement in time_statements:
        matches = re.finditer(r'\b' + re.escape(time_statement) + r'\b', text)
        for match in matches:
            entity = {
                "startTime": {
                    "value": match.group(),
                    "span": [match.start(), match.end()]
                }
            }
            entities["Time Statement"].append(entity)

    return entities
