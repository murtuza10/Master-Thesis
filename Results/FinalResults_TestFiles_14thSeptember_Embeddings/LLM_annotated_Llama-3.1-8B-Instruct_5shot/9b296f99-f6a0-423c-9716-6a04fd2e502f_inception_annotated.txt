import re
import json

def extract_entities(text):
    # Initialize entity lists
    crops = []
    soil = []
    location = []
    time_statement = []

    # Crops
    crop_species_pattern = r'\b(winter wheat|winter barley|sugar beet|maize|potatoes)\b'
    crop_varieties = re.findall(crop_species_pattern, text)
    for variety in crop_varieties:
        crops.append({
            "cropSpecies": {
                "value": variety,
                "span": [text.find(variety), text.find(variety) + len(variety)]
            }
        })

    # Soil
    soil_reference_groups = ["horse manure", "farmyard manure", "crop residues", "catch crop"]
    for group in soil_reference_groups:
        soil.append({
            "soilReferenceGroup": {
                "value": group,
                "span": [text.find(group), text.find(group) + len(group)]
            }
        })

    # Location
    location_pattern = r'\b(Germany|Speyer|Rinkenbergerhof|LUFA Speyer|Karlsruhe|Brandenburg|M채rkisch Oderland|M체ncheberg)\b'
    locations = re.findall(location_pattern, text)
    for location_name in locations:
        location.append({
            "country": {
                "value": location_name,
                "span": [text.find(location_name), text.find(location_name) + len(location_name)]
            }
        })

    # Time Statement
    time_statement_pattern = r'\b(1984|2009)\b'
    times = re.findall(time_statement_pattern, text)
    for time in times:
        time_statement.append({
            "year": {
                "value": time,
                "span": [text.find(time), text.find(time) + len(time)]
            }
        })

    # Create JSON output
    output = {
        "Crops": crops,
        "Soil": soil,
        "Location": location,
        "Time Statement": time_statement
    }

    return output
