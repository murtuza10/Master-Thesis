import json
import re

def extract_entities(text):
    # Initialize entities
    entities = {
        "Crops": [],
        "Soil": [],
        "Location": [],
        "Time Statement": []
    }

    # Crops
    crop_species = ["Bodenart"]
    for species in crop_species:
        matches = re.findall(species, text)
        for match in matches:
            entities["Crops"].append({
                "cropSpecies": {
                    "value": match,
                    "span": [text.find(match), text.find(match) + len(match)]
                }
            })

    # Soil
    soil_texture = ["Bodentyp", "Wasserhaushalts- Nährstoffansprache", "Kalkung"]
    soil_reference_group = []
    soil_depth = ["TRD", "nFK", "Entkalkungstiefe"]
    soil_bulk_density = []
    soil_ph = []
    soil_organic_carbon = ["organischer Kohlenstoffgehalt"]
    soil_available_nitrogen = []
    for texture in soil_texture:
        matches = re.findall(texture, text)
        for match in matches:
            entities["Soil"].append({
                "soilTexture": {
                    "value": match,
                    "span": [text.find(match), text.find(match) + len(match)]
                }
            })
    for depth in soil_depth:
        matches = re.findall(depth, text)
        for match in matches:
            entities["Soil"].append({
                "soilDepth": {
                    "value": match,
                    "span": [text.find(match), text.find(match) + len(match)]
                }
            })

    # Location
    location = ["Deutschland", "Bundeswaldinventur"]
    for loc in location:
        matches = re.findall(loc, text)
        for match in matches:
            entities["Location"].append({
                "country": {
                    "value": match,
                    "span": [text.find(match), text.find(match) + len(match)]
                }
            })

    # Time Statement
    start_date = re.search(r"start date: (\d{4})", text)
    end_date = re.search(r"end date: (\d{4})", text)
    duration = re.search(r"temporal coverage: start date: \d{4}, end date: \d{4}", text)
    if start_date:
        entities["Time Statement"].append({
            "startTime": {
                "value": start_date.group(1),
                "span": [text.find(start_date.group(0)), text.find(start_date.group(0)) + len(start_date.group(0))]
            }
        })
    if end_date:
        entities["Time Statement"].append({
            "endTime": {
                "value": end_date.group(1),
                "span": [text.find(end_date.group(0)), text.find(end_date.group(0)) + len(end_date.group(0))]
            }
        })
    if duration:
        entities["Time Statement"].append({
            "duration": {
                "value": "2012",
                "span": [text.find(duration.group(0)), text.find(duration.group(0)) + len(duration.group(0))]
            }
        })

    return json.dumps(entities, indent=4)

# Test the function
