### Code to extract entities from the text and return them in the JSON format
```python
import re
from typing import Dict, List

def extract_entities(text: str) -> Dict[str, List[Dict[str, str]]]:
    entities = {
        "Crops": [],
        "Soil": [],
        "Location": [],
        "Time Statement": []
    }

    # Crops
    crop_species = ["wheat", "barley", "sugar beet", "maize", "potatoes"]
    for species in crop_species:
        matches = re.finditer(species, text, re.IGNORECASE)
        for match in matches:
            start_index = match.start()
            end_index = match.end()
            entities["Crops"].append({
                "cropSpecies": {
                    "value": species.lower(),
                    "span": [start_index, end_index]
                }
            })

    # Soil
    soil_terms = ["soil organic carbon", "total nitrogen"]
    for term in soil_terms:
        matches = re.finditer(term, text, re.IGNORECASE)
        for match in matches:
            start_index = match.start()
            end_index = match.end()
            if term == "soil organic carbon":
                entities["Soil"].append({
                    "soilOrganicCarbon": {
                        "value": term.lower(),
                        "span": [start_index, end_index]
                    }
                })
            elif term == "total nitrogen":
                entities["Soil"].append({
                    "soilAvailableNitrogen": {
                        "value": term.lower(),
                        "span": [start_index, end_index]
                    }
                })

    # Location
    location_terms = ["Speyer", "Germany"]
    for term in location_terms:
        matches = re.finditer(term, text, re.IGNORECASE)
        for match in matches:
            start_index = match.start()
            end_index = match.end()
            if term == "Speyer":
                entities["Location"].append({
                    "city": {
                        "value": term.lower(),
                        "span": [start_index, end_index]
                    }
                })
            elif term == "Germany":
                entities["Location"].append({
                    "country": {
                        "value": term.lower(),
                        "span": [start_index, end_index]
                    }
                })

    return entities

# Text
```
